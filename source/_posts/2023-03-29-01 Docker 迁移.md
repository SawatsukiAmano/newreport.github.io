```bash
# 打包 volume
tar -zcvf volumes.tar.gz /var/lib/docker/volumes

# 将容器保存为镜像
docker commit container-id image-name 

# 解压 volumes
tar zxvf volumes.tar.gz /var/lib/docker

# 变更权限

# 加载镜像
cat image-name.tar | docker load 


```
